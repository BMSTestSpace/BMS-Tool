function showStep() { const step = tree.steps[currentStep]; const app = document.getElementById("app"); app.innerHTML = ""; // If it's a question if (step.question) { const question = document.createElement("h2"); question.innerText = step.question; app.appendChild(question); for (let answer in step.answers) { const btn = document.createElement("button"); btn.innerText = answer; btn.className = "btn"; btn.onclick = () => { currentStep = step.answers[answer]; showStep(); }; app.appendChild(btn); } } // If it's a result else { const resultBox = document.createElement("div"); resultBox.className = "result"; const result = document.createElement("h2"); result.innerText = "Diagnosis"; const diagnosisText = document.createElement("p"); diagnosisText.innerText = step.diagnosis; const actionText = document.createElement("p"); actionText.innerHTML = `Recommended Action: ${step.action}`; const restartBtn = document.createElement("button"); restartBtn.innerText = "Start Over"; restartBtn.className = "btn"; restartBtn.onclick = () => { currentStep = tree.start; showStep(); }; resultBox.appendChild(result); resultBox.appendChild(diagnosisText); resultBox.appendChild(actionText); resultBox.appendChild(restartBtn); app.appendChild(resultBox); } }
